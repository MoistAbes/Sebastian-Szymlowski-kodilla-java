SELECT I.SUMMARY, IL.NAME
FROM ISSUES I JOIN ISSUESlIST IL on IL.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, I.DESCIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUE_NUMBER
FROM ISSUES I JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0;
